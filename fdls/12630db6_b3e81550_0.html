本节课大纲：
一、模板包含
       <include file="完整模板文件名" />
               <include file="./Tpl/default/Public/header.html" />        //路径方式
               <include file="read" />                                //找本模块文件夹下的模板
               <include file="Public:header" />        //Tpl下的Public文件夹下的header，不需要控制器支持
               <include file="blue:User:read" />        //主题blue下的User模块下的read
               <include file="$tplName" />                
               <include file="header" title="ThinkPHP框架"keywords="开源WEB开发框架"/>
                       title和keywords为传递给即将被调用的模板的参数
                       在模板中变量用[变量]接受
               <include file='file1,file2' />        //可以引入多个文件
二、模板渲染(模板布局)
       1、自动开启模板渲染 设置配置文件
                       'LAYOUT_ON'=>true,//开启模板渲染，默认是false
                       准备一个模板渲染页面，在页面中使用{__CONTENT__}接受具体模板页面的内容
                       如果在摸一个具体模板中不希望使用渲染模板，可以在页首添加{__NOLAYOUT__}
       2、不开启自动模板渲染可以在每一个具体页面的页首添加
               <layout name='layout'/>
               在Tpl下建立layout.html模板
               准备一个模板渲染页面，在页面中使用{__CONTENT__}接受具体模板页面的内容
       3.使用技巧
               在渲染模板文件中也可以使用其他模板文件的内容
               <include file='Public:header'/>
               <body>
                       <p>这里是渲染页面！！！</p>
                       {__CONTENT__}
               </body>
       </html>

三、模板的继承        
       父模板中定义
               <block name="top">
               </block>
               <div style="border: 1px solid grey;background: #abcdef; height:10px"> 广告</div>
               <block name="footer">
               </block>

       子模板中实现继承，显示子模板中block中的内容
               <extend name="父模板名" />
               <block name="top">
                       会被显示的内容
               </block>

               <block name="footer">
                       ...
               </block>

       父模板中可以灵活的插入其它东西，父模板就类似最后的组装，子模板是具体的细节
