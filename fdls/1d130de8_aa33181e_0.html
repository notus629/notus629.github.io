删除数据的两种方式

    delete

    truncate



delete

    // 删除整表(表中所有数据）
    DB::table('student')->delete();

    // 带条件删除, id=15
    $num = DB::table('student')
        ->where('id', 15)
        ->delete();

    // 带条件删除, id>=13
    $num = DB::table('student')
        ->where('id', '>=', 13)
        ->delete();



truncate

    // 删除整个表的数据(同不带条件的 delete)，返回 null
    DB::table('student')->truncate();
