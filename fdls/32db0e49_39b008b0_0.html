在控制器中使用

    // 查询表中的所有记录，返回一个集合，集合下面包含一个数组
    // 数组的每个元素对应表的每行（每条记录），数组里面即对象（模型）
    // 每个元素的 attributes 属性包含了对应的一行的 N 个单元值
    // 实际返回的是一个对象
    $students = Student::all();

    // 根据主键进行查询, 查询id为1001的
    // 实际返回的是模型的对象
    $students = Student::find(1001);

    // 根据主键进行查找，没找到则抛出异常
    $studenets = Student::findOrFail(1006);



查询构造器在 Eloquent ORM 中的使用

    // 查询所有的
    $students = Student::get();

    // 查询一条记录
    $student = Student::where('id', '>', '1001')
        ->orderBy('age', 'desc')
        ->first();

    // 每次查两条
    echo '<pre>';
    Student::chunk(2, function($students){
        var_dump($students);
    });

    // 聚合函数
    $num = Student::count();
    $max = Student::where('id', '>', 1001)->max('age');
