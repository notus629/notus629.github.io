窗口页面尺寸和元素/滚动条位置

    获取和设置

    .height()            选取结果的高度

    .width()             选取结果的宽度

    .scrollLeft()        水平滚动条的位置

    .scrollTop()         垂直滚动条的位置


    返回对象，包含属性 top, left

    .offset()            获取或设置元素相对于文档对象(document)左上角的坐标

    .position()          仅获取元素相对于其偏移父元素(offset parent)的位置

<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>位置</title>
    <script type="text/javascript" src="js/jquery.js"></script>
    <script type="text/javascript">
        $(function(){
            var $offset = $('h1').offset();
            $('h1').after('<p>Top: ' + $offset.top + '</p>\n<p>Left: ' + $offset.left + '</p>');         
        });
    </script>
</head>
<body>
    <div>
        <h1>PHP</h1>
    </div>
</body>
</html>
