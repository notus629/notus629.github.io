<head>
    <meta charset="utf-8">
    <title>使用筛选</title>
    <script type="text/javascript" src="js/jquery.js"></script>
    <script type="text/javascript">
        $(function(){
            var $listItem = $('li');
            // 找到最后一个 class 属性包含 hot 的列表项
            $listItem.filter('.hot:last').removeClass('hot');
            // 查找 class 属性不包含 hot 的 li 元素
            // $listItem.not('.hot').addClass('complete');
            $('li:not(.hot)').addClass('complete');
            // 使用 .has() 方法查找其中含有 <em> 元素的 <li> 元素，然后添加 class 属性值 complete
            $listItem.has('em').addClass('complete');
 
            // 使用 is() 在包含 class 属性 hot 的元素前面加上 notus: 
            $listItem.each(function(){
                var $this = $(this);
                if($this.is('.hot')){
                    $this.prepend('Notus: ');
                }
            });
 
            // 使用 :contains() 查找包含文本 PHP 的 li 元素，在其后加上 (best language)
            $('li:contains(PHP)').append(' (best language)')
 
        });
    </script>
    <style type="text/css">
        .hot {
            color: red;
        }
        .cool {
            color: blue;
        }
        .complete{
            background-color: #ccc;
        }
    </style>
