post

    向服务器发送数据

    methods: {
        get: function(){
            // 获取普通文本数据
            this.$http.post('post.php', {
                a:1,
                b:19
            }, {
                emulateJSON:true
            }).then(function(res){                 // $http 里头也是 ajax 对象， get 参数： url地址
                alert(res.data);                                        // then (成功函数，失败函数), res.status, res.data 数据
            }, function(){
                alert(res.status);
            });
        } 

    }

    <input type="button" @click="get()" value="click">
