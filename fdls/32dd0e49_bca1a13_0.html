使用模型的 Create 方法新増数据

    // 先在模型中指定允许批量赋值的字段
    protected $fillable = ['name', 'age'];

    // 再用 create 新増数据
    $student = Student::create(
        ['name' => 'imooc', 'age' => 18]
    );



指定不允许批量赋值的字段

    protected $guarded = [...];



firstOrCreate()

    // 以属性查找用户，若没有则新増，并取得新的实例
    $student = Student::firstOrCreate(
        ['name' => 'immoc']
    );



firstOrNew()

    // 以属性查找用户，若没有则建立新的实例，若要保存，自己调用 save
    $student = Student::firstOrNew(
        ['name' => 'immocss']
    );
    $bool = $student->save();
