本节课大纲：
一、导入CSS和JS文件      
       1. 传统方式
               css link:  <link rel='stylesheet' type='text/css' href='__PUBLIC__/Css/test.css'/>
                 js  scr:        <script src='__PUBLIC__/Js/test.js'></script>
       2. import
               <import type='js' file='Js.test' />
                       //导入Public文件夹下面的Js目录中的test.js文件
                       //import标签可以省略type属性，默认就是js的(type可省)
               <import type='css' file='Css.test' />
               可以更改默认文件夹 设置basepath属性
               <import type='js' file='Js.my' basepath='./Other'/>
                       ./指项目文件夹，即index.php所在目录
       3. load
               //方法可以自动检测导入的文件类型
               <load file='__PUBLIC__/Js/test.js' />

二、分支结构
       1、if        
               <if condition= ' $sex eq "男" '>        //$sex为后台分配过来的变量，此时外面不加{}符
                       男人是泥巴做的
                       <else />
                       女人是水做的
               </if>
               <if condition='$age lt 18'>                //直接写<不行
                       未成年
                       <elseif  condition='$age eq 18'/>        //直接写==不行
                       青春年少
                       <else />
                       成年
               </if>
               比较运算
                       >    gt
                       <    lt
                       ==   eq
                       <=   elt
                       >=   egt
                       !=   neq
                       ===   heq
                       !==   nheq
       2. switch case 结构
               <switch name='number'>        //去掉$符
                       <case value='1'>一个和尚挑水吃</case>
                       <case value='2'>两个和尚台水吃</case>
                       <case value='3'>三个和尚没水吃</case>
                       <default/> 这里是默认值
               </switch>

三、循环结构
       1.for
               <table border='1' width='500'>
                       <for start='10' end='00' name='j' step='-2' comparison='gt'>        
                               //默认name值为i即，变量为$i
                               //step：步进值，默认为1
                               //comparison：判断条件，默认为lt
                               <tr><td>{$j}</td><td>abc</td></tr>
                       </for>
               </table>

       2.volist
               <volist name='list' id='v' offset="1" length="3">
                       //name: 从模块拿到的数组变量
                       //id: 数组解析出的每个变量依次每轮赋给v
                       // 从offset开始，输出length个变量
                       {$v.username}<br/>
               </volist>
       3.foreach
               <foreach name='list' item='v' key='k'>
                       //name: 传递过来的数组
                       // item: 数组对应的vaule值
                       // key: 数组对应的键
                       {$k}-------{$v}<br/>
               </foreach>

四、特殊标签
       1、比较标签
               eq或者 equal 等于
                       <eq name='n' value='10'>                //n为传递过来的变量，不带$符
                               等于10
                               <else />
                               不等于10
                       </eq>
               neq        或者notequal 不等于
               gt        大于
               egt        大于等于
               lt        小于
               elt        小于等于
               heq        恒等于
               nheq        不恒等于
       2.范围标签
               in      
                       <in name='n' value='9,10,11,12'>
                               在这些数字里面
                               <else/>
                               不在这些数字的范围内
                       </in>
                       <notin name='n' value='9,10,11,12'>
                               在这些数字里面
                               <else/>
                               不在这些数字的范围内
                       </notin>
               between
                       <notbetween name='n' value='1,10'>
                               {$n}不在1-10之间
                               <else/>
                               {$n}不在1到10之间
                       </notbetween>
       3.present
               标签来判断模板变量是否已经赋值，
               <present name='m'>m有赋值<else/>m没有赋值</present>
       4.Empty
               empty标签判断模板变量是否为空，
               <empty name='n'>n为空赋值<else/>n有值</empty>
                       //n 是否为空值（null)
       5.Defined
               判断常量是否已经定义
       6.Define
               在模板中定义常量
       7.Assign
               模板中变量赋值
                               
               
五、其他标签使用
       1、在模板中直接使用PHP代码
               <php> echo "我是赵桐正" </php>
               //不建议使用，建议前台只使用标签
               关闭原生php标签（即<?php ... ?>）: 设置TMPL_DENY_PHP为true, <php>标签仍能用
               <php>标签里不能再嵌入其它标签
       2、建议更改左右定界符
               在配置文件中改变
                       'TMPL_L_DELIM'=>'<{', //修改左定界符
                       'TMPL_R_DELIM'=>'}>', //修改右定界符
