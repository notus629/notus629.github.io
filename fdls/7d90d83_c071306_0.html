Smarty

    
    引入    

        require ('../smarty/Smarty.class.php');
        

    创建
        
        $smarty = new Smarty();


    Smarty 的“五配置两方法”

        // 五配置
        $smarty->left_delimiter = "{";       // 左定界符
        $smarty->right_delimiter = "}"; 
        $smarty->template_dir = "tpl";       // html模板的地址
        $smarty->comile_dir = "template_c";  // 模板编译生成的文件
        $marty->cache_dir = "cache";         // 缓存

        //以下是开启缓存的另两个配置。因为通常不用smarty的缓存机制，所以此项为了解
        $smarty->caching = true;             //开启缓存
        $smarty->cache_lifetime = 120;       // 缓存时间

        // 两方法 
        $smarty->assign('articletitle', '文章标题');    // 往模板中注册变量
        $smarty->display('test.tpl');        // 展示模板
