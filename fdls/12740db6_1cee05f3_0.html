<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>v-on</title>
    <script src="vue.js"></script>
    <script>
        window.onload = function(){
            new Vue({               // json 格式
                el: '#box',         // 选择父元素
                data:{              // vue 数据
                    myArr: [],
                    postfix: '',
                },
                methods:{
                    addArrList: function(){
                        this.myArr.push('hi, Leo');
                    },
                    deleteArrList: function(){
                        this.myArr.pop();
                    }
                }
 
            });
        }
    </script>
</head>
<body>
    <div id="box">
        <!--事件处理方式1-->
        <button v-on:click="addArrList()">增加列表</button>
        <button v-on:click="deleteArrList()">删除列表</button>
        <ul>
            <!--默认列表中元素重复，重复时使用 track-by="$index"-->
            <li v-for="value in myArr" track-by="$index">
                {{$index}} - {{value}} {{postfix}}
            </li>
        </ul>
        <hr>
        <!--事件处理方式2-->
        <button v-on:click="postfix='notus'">Add notus</button>
        <button v-on:click="postfix=''">Delete notus</button>
    </div>
</body>
</html>
