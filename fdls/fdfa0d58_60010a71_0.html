vue 属性 - 示例

<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>vue attribute</title>
    <script src="lib/vue.js"></script>
    <script>
        onload = function(){
            var vm = new Vue({
                el: '#box',
                data: {
                    url: 'https://ss0.bdstatic.com/5aV1bjqh_Q23odCf/static/superman/img/logo/bd_logo1_31bdc765.png',
                    height: "50px",
                    width: '200px',
                    title: '好样的'
                },
            });
        };
    </script>
</head>
<body>
    <div id="box">
        <img src="{{url}}" alt="bd">
        <br><hr><br>
        <img v-bind:src="url" alt="bd2">
        <br><hr><br>
        <img :src="url" :height="height" alt="bd3" :title="title" :width="width">
    </div>
</body>
</html>
