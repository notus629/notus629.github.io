聚合函数

    count()

    max()

    min()

    avg()

    sum()



count()

    // 返回记录数
    $num = DB::table('student')->count();



max(), min()

    // 返回最大值
    $max = DB::table('student')->max('age');

    // 返回最小值
    $min = DB::table('student')->min('age');



avg(), sum()

    // 返回平均值
    $avg = DB::table('student')->avg('age');

    // 返回值的和
    $sum = DB::table('student')->sum('age');
