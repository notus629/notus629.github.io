交互

    vue 框架本身不支持交互(leo, 好像新的版本有了？？)


    引入 vue-resource （vue-resource.js）

        <script src="vue.js"></script>
        <script src="vue-resource.js"></script>


    交互方式: get, post, jsonp



get
    
    获取数据

    methods: {
        get: function(){
            // 获取普通文本数据
            this.$http.get('a.txt').then(function(res){                 // $http 里头也是 ajax 对象， get 参数： url地址
                alert(res.data);                                        // then (成功函数，失败函数), res.status, res.data 数据
            }, function(){
                alert(res.status);
            });
        } 

    }

    <input type="button" @click="get()" value="click">


    向服务器发送数据

    methods: {
        get: function(){
            // 获取普通文本数据

            }, function(){
                alert(res.status);
            });
        } 

    }

    <input type="button" @click="get()" value="click">



post

    向服务器发送数据

    methods: {
        get: function(){
            // 获取普通文本数据
            this.$http.post('post.php', {
                a:1,
                b:19
            }, {
                emulateJSON:true
            }).then(function(res){                 // $http 里头也是 ajax 对象， get 参数： url地址
                alert(res.data);                                        // then (成功函数，失败函数), res.status, res.data 数据
            }, function(){
                alert(res.status);
            });
        } 

    }

    <input type="button" @click="get()" value="click">

    
jsonp

    methods: {
        get: function(){
            // 获取普通文本数据
            this.$http.jsonp('xxxxxx', {
                word:'a'
            }, {
            }).then(function(res){                 // $http 里头也是 ajax 对象， get 参数： url地址
                alert(res.data.s);                 // then (成功函数，失败函数), res.status, res.data 数据
            }, function(){
                alert(res.status);
            });
        } 

    }
