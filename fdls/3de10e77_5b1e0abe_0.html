class 特性用法

    // 下面各示例代码的公共部分，如下
    <style>
        .class1{
            color: red;
        }

        .class2{
            background-color: blue;
        }
    </style>



class 用法 1， 数组方式

    // 此处 class1, class2 为原生的 class 特性值
    data: {
        cls1: 'class1',        
        cls2: 'class2',
    }

    // cls1, cls2 是 vue 数据
    :class="[cls1, cls2]"    



class 用法 2, json 属性方式

    // 直接使用原生的 class 特性值, 此例表示 cls1 生效，cls2 不生效
    :class="{cls1: true, cls2: false}"        



class 用法 3， 两种方式组合

    data: {
        class1: true,        
        class2: false,
    }

    // cls1 生效，cls2 不生效
    :class="{cls1: class1, cls2: class2}"    



class 用法 4， vue-json 官方推荐

    // 此处 cls1, cls2 为原生的 class 特性值
    data: {
        myjson:{
            cls1: true,        
            cls2: false,
        }
    }

    // cls1 生效，cls2 不生效
    :class="myjson"
