style 特性用法


    方式 1 - JSON 特性形式           

        :style="{color:red}"


    方式 2 - 数组形式                

        data: {
            c: {color: 'red'},                      // 复合样式采用驼峰命名法
            b: {backgroundColor: 'blue'},           // 注意此处, 并非 background-color, Json 的原因，不能有 '-' 字符
        }

        :style="[c, b]"
  
      
    方式 3 , vue JSON 形式，官方推荐  

        data: {
            myjson: {
                color: 'red',
                backgroundColor: 'blue',
            }
        }

        :style="myjson"
