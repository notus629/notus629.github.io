    <meta charset="utf-8">
    <title>jQuery 事件对象</title>
    <script type="text/javascript" src="./js/jquery-3.1.1.min.js"></script>
    <script type="text/javascript">
        $(function(){
            $('li').on('click', function(e){
                $('li>span').remove();
 
                // 创建新的日期对象, 并将其时间设置为点击事件发生的时间
                var date = new Date();
                // timeStamp 在 chrome (Version 58.0.3029.81 (64-bit))中为事件发生时刻距页面生成时间的毫秒数，firefox （2017-4-27）计算不正确
                date.setTime(e.timeStamp);
                // 将事件发生的时间转换为可读的日期
                var clickDate = date.toDateString();
 
                $(this).append('<span> ' + clickDate + ' ' + e.type + ' </span>');
 
            });
 
        });
    </script>
