    <script type="text/javascript" src="js/jquery-3.1.1.min.js"></script>
    <script type="text/javascript">
        $(function(){
            // h1 隐藏后淡出
            $('h1').hide().slideDown();
     
            var $list = $('li');
 
            // 按顺序淡出所有列表(每隔700ms)
            $list.each(function(index){
                $(this).hide().delay(700*(index+1)).fadeIn(700);
            });
 
            // 单击元素淡出
            $list.on('click', function(){
                $(this).fadeOut(700);
            })
 
        });
    </script>
