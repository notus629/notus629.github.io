v-for

    用于循环

    {{$index}}  索引编号

    {{$key}}    键


    arr 为数组格式数据

        // 索引 {{$index}
        <li v-for="value in arr">
            {{value}} {{$index}}
        </li>


    json 为 json 格式数据

        // json key {{$key}}, 索引 {{$index}}
        <li v-for="value in arr">
            {{value}} {{$index}} {{$key}}
        </li>

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>v-for</title>
    <script src="vue.js"></script>
    <script>
        window.onload = function(){
            new Vue({               // json 格式
                el: '#box',         // 选择父元素
                data:{              // vue 数据
                    arr: ['apple', 'pear', 'banana', 'orange'],
                    json: {
                        a: 'apple',
                        b: 'pear',
                        c: 'banana',
                        d: 'orange'
                    }
                }
            });
        }
    </script>
</head>
<body>
    <div id="box">
        <!--循环一：数组-->
        <ul>
            <li v-for="value in arr">
                {{value}} - {{$index}}
            </li>
        </ul>
 
        <!--循环二: json-->
        <ul>
            <li v-for="value in json">
                {{value}} - {{$index}} - {{$key}}
            </li>
        </ul>
 
        <!--循环三: json (k,v) 方式-->
        <ul>
            <li v-for="(k,v) in json">
                k:{{k}}, v:{{v}}, index:{{$index}}, key:{{$key}}
            </li>
        </ul>
    </div>
</body>
</html>

  • k:a, v:apple, index:0, key:a
  • k:b, v:pear, index:1, key:b
  • k:c, v:banana, index:2, key:c
  • k:d, v:orange, index:3, key:d
