class 特性用法 - 示例

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>vue class usage</title>
    <script src="lib/vue.js"></script>
    <script>
        onload = function(){
            var vm = new Vue({
                el: '#box',
                data: {
                    cls1: 'class1',
                    cls2: 'class2',
                    c1: false,
                    c2: true,
                    classJson: {
                        class1: true,
                        class2: true,
                    }
                }
            });
        };
    </script>
    <style>
        .class1 {
            color: red;
        }
        .class2 {
            background-color: whitesmoke;
        }
    </style>
</head>
<body>
<div id="box">
    <!--用法1, 数组形式    -->
    <h1 :class="[cls1, cls2]">Hello Notus</h1>
    <!--用法2, Json 特性形式-->
    <br><hr><br>
    <h1 :class="{class1: true, class2: false}">Hello Notus</h1>
    <!--用法3，混合形式-->
    <br><hr><br>
    <h1 :class="{class1: c1, class2: c2}">Hello Notus</h1>
    <!--用法4，vue Json 方式，推荐-->
    <br><hr><br>
    <h1 :class="classJson">Hello Notus</h1>
</div>
</body>
</html>
