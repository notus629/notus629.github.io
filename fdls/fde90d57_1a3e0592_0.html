vue 属性

    v-bind:src=""


    简写  :src="", 推荐

        :width, :title, :height 等类似


    data:{
        url: '...',
    }


    <!-- 方式一： 会出来，但会报 404 的错误  -->
    <img src="{{url}}" alt="">     


    <!-- 方式二：vue 方式， 不会发 404 请求 -->
    <img v-bind:src="url" alt="">  

    <img :src="url" alt="" :width="w">
