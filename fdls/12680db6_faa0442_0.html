class 和 style

    :class=""

    :style=""


    class 用法 1， 数组方式

        data: {
            class1: 'cls1',        // 此处 cls1, cls2 为原生的 class 属性值
            class2: 'cls1',
        }

        :class="[class1, class2]"    // class1, class2 是 vue 数据


    class 用法 2, json 方式

        :class="{cls1: true, cls2: false}"        // 直接使用原生的 class 属性值, cls1 生效，cls2 不生效


    class 用法 3， 组合

        data: {
            class1: true,        
            class2: false,
        }

        :class="{cls1: class1, cls2: class2}"    // cls1 生效，cls2 不生效


    class 用法 4， 官方推荐

        data: {
            myjson:{
                cls1: true,        // 此处 cls1, cls2 为原生的 class 属性值
                cls2: false,
            }
        }

        :class="myjson"    // cls1 生效，cls2 不生效



style


    方式 1 

        :style="{color:red}"


    方式 2

        data: {
            c: {color: 'red'},                      // 复合样式采用驼峰命名法
            b: {backgroundColor: 'blue'},           // 注意此处, 并非 background-color,json的原因，不能有 '-' ??
        }

        :style="[c, b]"
  
      
    方式 3 , 官方推荐

        data: {
            myjson: {
                color: 'red',
                backgroundColor: 'blue',
            }
        }

        :style="myjson"
