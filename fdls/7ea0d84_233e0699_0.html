在smarty里面输出赋值进来的变量（数组、类）

    php文件中

        $arr = array('title'=>'smarty的学习', 'author'=>'小明');
        $arr2 = array('articlecontent'=>array('title'=>'smarty的学习', 'author'=>'小明'));
        $smarty->assign('arr', $arr);
        $smarty->assign('arr2', $arr);
        $smarty->display('test.tpl');

    
    模板里输出
            {$arr.title} {$arr.author}   //方法1
            {$arr['title']} {$arr['author']}  //方法2
            {$arr2['articlecontent']['title']}  //多维数组， ？$arr2.articlecontent.title?

    Smarty中，使用php内置函数和自定义函数，还可以自定义插件
    
        使用php内置函数
          {"Y-m-d"|date:$time}
          {'d'|str_replace:'h':$str}
          
        自定义的函数，registerPlugin
            function test($params){
                $p1 = $params['p1'];
                $p2 = $params['p2'];
                return '传入的参数1值为'.$p1.'，传入的参数2值为'.$p2;
            }
            $smarty->registerPlugin('function', 'f_test', 'test');
        
          使用 
              {f_test p1='abc' p2='edf'}    //函数的两个参数（属性）p1, p2，会被打包成一个数组params传递给函数

    Smarty的插件
        本质上是function函数
        
        类型：functions 函数插件, modifiers 修饰插件, block functions区块函数插件
    
        制作、使用插件：
        
            方法1. 使用registerPlugin方法注册写好的自定义函数
            方法2. 将写好的插件放入Smarty解压目录的lib目录下的plugins目录里
            方法3. php内置，可以自动以修饰插件（变量调节器插件）的形式在模板里使用
              
        编写function插件
            在plugins目录，新建function.test.php文件
                <?php
                    function smarty_function_test($param){ //smarty_function_插件名
                        // $params的格式：array('参数1'=>'参数值'，'参数2'=>'参数值')
                        // $参数1 = $params['参数1'];
                        // $参数2 = $params['参数2'];
                        $width = $params['width'];
                        $height = $params['height'];
                        $area = $width*$height;
                        return $area;
                    }
                ?>
        

        编写modifier插件(即变量调节器)
            在plugins目录，新建modifier.test1.php文件
                <?php
                    function smarty_modifier_php($utime, $format){
                        return date($format, $utime);
                    }
                ?>
        
        
        编写区块插件(插件名不允许重复)
        
            block.test2.php
