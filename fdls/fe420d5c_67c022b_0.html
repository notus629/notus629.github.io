jsonp

    methods: {
        get: function(){
            // 获取普通文本数据
            this.$http.jsonp('xxxxxx', {
                word:'a'
            }, {
            }).then(function(res){                 // $http 里头也是 ajax 对象， get 参数： url地址
                alert(res.data.s);                 // then (成功函数，失败函数), res.status, res.data 数据
            }, function(){
                alert(res.status);
            });
        } 

    }

    <input type="button" @click="get()" value="click">


    methods: {
        get: function(){

                alert(res.data.s);                 // then (成功函数，失败函数), res.status, res.data 数据
            }, function(){
                alert(res.status);
            });
        } 

    }
