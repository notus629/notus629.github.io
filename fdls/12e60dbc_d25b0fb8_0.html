style 特性用法 - 示例

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>style 特性</title>
    <script src="lib/vue.js"></script>
    <script>
        onload=function(){
            new Vue({
                el: '#box',
                data: {
                    mycolor: {color: 'red'},
                    mybgcolor: {backgroundColor: 'grey'},
                    clr: 'red',
                    bkclr: 'grey',
                    myjson: {
                        color: 'red',
                        backgroundColor: 'grey',
                    }
                },
            });
        }
    </script>
</head>
<body>
    <div id="box">
        <!--方式一 数组形式-->
        <h1 :style="[mycolor, mybgcolor]">Hi Notus</h1>
        <!--方式二 json 特性形式-->
        <h1 :style="{color:clr, backgroundColor:bkclr}">Hi Notus</h1>
        <!--方式三 vue json 形式-->
        <h1 :style="myjson">Hi Notus</h1>
    </div>
</body>
</html>
