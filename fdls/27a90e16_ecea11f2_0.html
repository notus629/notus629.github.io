laravel


    app          包含了应用程序的核心代码：自己编辑的代码
        Http/
            Controllers/  控制器（C）目录
            routes.php    路由


    bootstrap    包含系统启动，和自己加载配置的文件


    config       包含所有应用程序的配置文件
        cache.php        缓存
        database.php     数据库


    database     包含了数据库迁移与填充文件


    public       入口、项目的静态资源文件（项目的图片、js、css）
        index.php        入口文件


    resources    包含视图及原始的资源文件
        views/   视图(MVC中的V， 视图都在写在这里）
            welcome.blade.php    输出模板


    storage      编译后的模板文件，基于文件的 session, 文件缓存，日志 和其它框架产生的文件
        framework
            views        编译后产生的文件，显示"Laravel 5"的文件就在这里
        logs


    tests        测试，单元测试目录，存放单元测试代码


    vendor       composer 下载的依赖模块
