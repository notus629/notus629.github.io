PHP中

    调用模块中的模板文件

        include $this->template('activity');    // 在site.php的主类中使用，activity位于模块的template目录下
                                                // 若是手机端，应位于template/mobile下

    调用系统模板文件(在主系统PHP代码中）
        <?php
            template('user/login');             // 将渲染 web/themes/default/user/login.html文件
                                                // app端渲染 app/themes/default/xxx/yyy.thml文件


模板中

    {template 'common/header'}                  // 系统的模板的引用的方法，需要添加目录信息
                                                // 位于：web/themes/default/common                                       
  
    {template 'header'}                         // 模块中的使用方法，不要添加目录信息
