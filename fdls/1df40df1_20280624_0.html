        方法二：利用 javascript 来实现

            1. 点击按钮，给按钮一个绑定事件

            2. 点击背景幕时，不让模态框隐藏

            3. 将模态框的内容写到外部的文件，然后再加载进来



模态框显示事件

    模态框显示前  show.bs.modal

    模态框显示后  shown.bs.modal

    模态框关闭前  hide.bs.modal

    模态框关闭后  hidden.bs.modal

<script>
    $(function(){
        $('#openModal').click(function(){
            //$('#mymodal').modal('show');
            $('#mymodal').modal({
                'backdrop': 'static',       // 点击背景幕时模态框不消失
                //'show':true,                // 一上来就显示
                //'remote':'remote.html',     // 表示加载外部的内容进来（即把模态框的modal-content部分放入外部 html）
            });
        });
 
        $('#mymodal').on('show.bs.modal', function(){
            alert('模态框即将显示!');
        });
        $('#mymodal').on('shown.bs.modal', function(){
            alert('模态框已经显示出来了!');
        });
        $('#mymodal').on('hide.bs.modal', function(){
            alert('模态框即将关闭!');
        });
        $('#mymodal').on('hidden.bs.modal', function(){
            alert('模态框已经关闭!');
        });
 
    });
</script>
