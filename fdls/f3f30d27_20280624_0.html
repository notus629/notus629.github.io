模态框



模态框属性

    .modal                          必须

    role="dialog"                   必须

    .fade                           淡入效果，删掉则模态框会直接出现

    id="myModal"                    id 值

    tabindex="-1"                   习惯用法

    aria-labelledby="myModalLabel"  对应模态框标题的 id

    data-dismiss="modal"            关闭模态框



开关按钮属性

    data-toggle="modal"             必须，表示弹出模态框

    data-target="#myModal"          对应的被弹出模态框的 id



示例

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>模态框示例</title>
    <link rel="stylesheet" href="lib/bootstrap.css">
    <script src="lib/jquery.js"></script>
    <script src="lib/bootstrap.js"></script>

</head>
<body>
<div class="container">
    <!--模态框弹出按钮-->
    <button class="btn btn-danger" data-toggle="modal" data-target="#myModal">删除全部</button>

    <!--模态框-->
    <div class="modal fade" role="dialog" id="myModal" tabindex="-1" aria-labelledby="myModelLabel">
        <div class="modal-dialog" role="document">
            <div class="modal-content">
                <!--模态框标题栏-->
                <div class="modal-header">
                    <button type="button" class="close" data-dismiss="modal" aria-label="close">
                        <span aria-hidden="true">&times;</span>
                    </button>
                    <h4 class="modal-title" id="myModelLabel">确定删除?</h4>
                </div>
                <!--模态框内容-->
                <div class="modal-body text-right">
                    <button type="button" class="btn btn-primary btn-sm" data-dismiss="modal">取消</button>
                    <button type="button" class="btn btn-danger btn-sm">确定</button>
                </div>
            </div>
        </div>
    </div>
</div>
</body>
</html>
