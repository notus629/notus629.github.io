    实质： v-show="true" 即 display:"none";

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>v-on</title>
    <script src="vue.js"></script>
    <script>
        window.onload = function(){
            new Vue({               // json 格式
                el: '#box',         // 选择父元素
                data:{              // vue 数据
                    msg: '',
                    btnText: '显示',
                    show: false,
                },
                methods: {                      // 方法
                    toggle: function() {
                        if(this.show == false){
                            this.show = true;
                            this.btnText = '隐藏';
                        } else {
                            this.show = false;
                            this.btnText = '显示';
                        }
                    },
                },
            });
        }
    </script>
</head>
<body>
    <div id="box">
        <!--“显示”和“隐藏”开关-->
        <button v-on:click="toggle()">{{btnText}}</button>
        <div style="width:100px; height:100px; background-color: red;" v-show="show"></div>
    </div>
</body>
</html>
