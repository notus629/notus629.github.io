        所有列：select *

        where 子句运算符        
                =, >, <, >=, <=, !=, <>, 
                IS NOT NULL, IS NULL
                BETWEEN
                IN, NOT IN
                LIKE, NOT LIKE                   // 模式匹配， % 任意数量字符，_ 一个字符
                REGEXP/RLIKE                     // 正则表达式

                逻辑运算：and, or

        关联操作
                表名称间逗号， INNER JOIN, CROSS JOIN             // 完全关联，即笛卡尔积
                左关联                                           // 右边表没匹配行，结果也会增加一行，右边为NULL
                        from customer left join orders 
                        on customer.customerid = orders.customerid;
                        
                        or using (customerid)

        MySQL 关联总结
                笛卡尔积、完全关联、交叉关联
                内部关联
                等价关联
                左关联、右关联

        as 子句                  // 表、列的别名

        ORDER BY子句            // 以特定的顺序获取数据
                ASC、DESC关键字

        合计函数
                AVG（列）
                COUNT（项目）
                MIN(列）
                MAX(列）
                STD(列）
                STDDEV(列）
                SUM(列）

        GROUP BY 子句           // 按分组合计

        HAVING 子句             // 测试一个合计的结果

        LIMIT 子句              // 限制返回的行数

        子查询
                基本子查询
                        ANY, IN, SOME, ALL

                关联子查询
                        exists, not exists, 
                        外部查询，内部查询

        更新数据库记录
UPDATE  [LOW_PRIORITY]  [IGNORE]  tablename
SET  column1 = expression1, column2 = expression2, ...
[WHERE  condition]
[ORDER  BY  order_criteria]
[LIMIT number]

        创建后修改表
                ALTER  TABLE  [IGNORE]  table_name  alteration  [, alteration ... ]
                
        删除数据库中的记录
          DELETE  [ LOW_PRIORITY ]  [ QUICK ]  [ IGNORE ]  FROM  table
        [WHERE  condition ]
        [ ORDER  BY  order_cols ]
          [ LIMIT  number ]              

                delete from table_name;                         // 删除表中所有行
