<?php
 
$mem = new Memcache();
 
if ( !$mem->connect('127.0.0.1', 11211) ){
    die('连接 memcached 服务器失败！');
} else {
    echo ('连接 memcached 服务器成功!<br />');
}
 
// 添加字符串
if( $mem->add('leo', 'happy', MEMCACHE_COMPRESSED, 60) ) {
    echo ('保存字符串成功！<br />');
} else {
    echo ('保存字符串失败! <br />');
}
 
// 设置数值
if( $mem->set('leon', 1875, MEMCACHE_COMPRESSED, 60) ) {
    echo ('设置数值成功！<br />');
} else {
    echo ('设置数值失败! <br />');
}
 
// 设置数组
$arr = ["小五", "老七"];
if( $mem->set('leoa', $arr, MEMCACHE_COMPRESSED, 60) ) {
    echo ('设置数组成功！<br />');
} else {
    echo ('设置数组失败! <br />');
}
 
// 设置对象
class Dog {
    public $name;
    public $age;
    public function __construct($name, $age){
        $this->name = $name;
        $this->age = $age;
    }
}
 
$dog = new Dog('xinxin', 0.3);
 
if( $mem->set('leod', json_encode($dog), MEMCACHE_COMPRESSED, 60) ) {
    echo ('设置对象成功！<br />');
} else {
    echo ('设置对象失败! <br />');
}
 
// 获取值
if( $var = $mem->get('leod') ){
    echo '获取到 dog 的值为：<br />';
    var_dump( json_decode($var) );
} else {
    echo '获取值失败!<br/>';
}
 
// 删除所有值
if( $mem->flush() ){
    echo '删除所有缓存成功！<br />';
}
