<?php
 
class File{
 
    private $_dir;
 
    const EXT='.txt';
 
    public function __construct(){
        $this->_dir = dirname(__FILE__).'/files/';
    }
 
    public function cacheData($key, $value='', $path=''){
        $filename = $this->_dir.$path.$key.self::EXT;
 
        // 若 $value 有值，即要生成缓存，将 value 值写入缓存
        if($value !== ''){
 
            // 若 $value 为 NULL，则删除缓存 
            if( is_null($value) ){
                return @unlink($filename);
            }
 
            $dir = dirname($filename);
            if( !is_dir($dir) ){
                mkdir($dir, 0777);
            }
            return file_put_contents( $filename, json_encode($value) );
        }
 
        // $value 为空则获取缓存
        if(!is_file($filename)){
            return FALSE;
        } else {
            return json_decode( file_get_contents($filename), true );
        }
    }
 
}

<?php
 
require_once('./file.php');
 
$data = [
    'id' => 1,
    'name' => 'singwa',
    'type' => [4, 5, 6],
    'test' => [1, 45, 67=>[123, 'tsysa'] ]
];
 
$file = new File();
 
// 生成缓存
if ( $file->cacheData('index_mk_cache', $data) ){
    echo "Success!";
} else {
    echo "Error!";
}
 
// 获取缓存
if ( $file->cacheData('index_mk_cache') ){
    var_dump( $file->cacheData('index_mk_cache') ); exit;
} else {
    echo "Error!";
}
 
// 删除缓存
if ( $file->cacheData('index_mk_cache', null) ){
    echo "delete success";
} else {
    echo "delete error!";
}
