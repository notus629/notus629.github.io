中文存储到mysql乱码

        创建 mysql 数据库和表时，设置为 utf8编码，和utf8_gernal_ci校验

        php 文件是 utf8 格式

        html 是 utf8

        php 以 utf8 连接 mysql，如下设置
            // 设置连接的默认编码
            $conn->set_charset('utf8');
