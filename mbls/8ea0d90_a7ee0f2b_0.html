属性值的顺序总结

    

margin （padding 同）

    // 上，右，下，左 (即从上开始，顺时针顺序)
    margin:10px 5px 15px 20px;

    // 上，右/左，下
    margin:10px 5px 15px;

    // 上/下，右/左
    margin:10px 5px;



border

    // 分别（按顺序）：border-width, border-style, 和 border-color.
    border:5px solid red;



background 

    // background-color, background-image, background-size, background-repeat, background-origin, background-clip, background-attachment, 和 background-position. 有些可省略
    background: #00ff00 url('smiley.gif') no-repeat fixed center;
