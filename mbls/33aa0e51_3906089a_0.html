文件属性

    fileattime($file);                 // 返回文件最近被访问的时间戳

    filemtime($file);                  // 返回文件最近被修改的时间戳

    fileowner($file);                  // 返回文件的 uid （用户标识），不（可靠）支持 Windows

    filegroup($file);                  // 返回文件的 gid （组标识），不（可靠）支持 Windows

    posix_getpwuid($uid);              // 把 uid 转变成容易理解的名字，不（可靠）支持 Windows

    posix_getgrgid($gid);              // 把 gid 转变成容易理解的名字，不（可靠）支持 Windows

    fileperms($file);                  // 返回文件的权限码

    filetype($file);                   // 返回关于所检查文件的类型信息（fifo, char, dir, block etc.)

    filesize($file);                   // 返回文件的大小（Byte)

    stat();                            // 文件属性信息

    lstat();                           // 用于符号链接文件的属性信息

    clearstatcache();                  // 属性信息变化后，再检查时，得清除原有缓存


        
文件的属性是否判断

    is_dir();                          // 检查指定属性，返回 true 或 false
    is_executable();
    is_file();
    is_link();
    is_readable();
    is_writable();



更改文件属性(linux)

    chgrp(file, group);                // 更改文件的组          
    chmod(file, permissions);          // 修改文件的权限
    chown(file, user);                 // 修改文件的所有者
