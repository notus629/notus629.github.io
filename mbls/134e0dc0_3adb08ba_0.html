常见属性

    imgElement.src                          // 可以给图片src属性赋值

    someElement.value                       // 得到值，特别对于文本框、密码框等。也可以赋值

    event.keyCode                           // 按键的keycode（键码）, event为事件函数传递过来的事件对象

    window.location                         // 直接可以赋值，则把用户带到一个新地址
      
    windows.parent                          // 返回父窗口
        
    screen.width                            // 屏幕宽度

    window.location                         当前页面的URL

    document.URL                            当前页面的URL

    document.title                          web 页面上 <title></title> 标签之间的内容

    document.lastmodified                   页面最后被修改的日期        

    Math.PI;                                pi 的值



style 对象
    
    使用 style 对象的属性的语法

        document.getElementById("id").style.property = "值";
            
        e.g. Object.style.visibility = visible|hidden|collapse;



数组属性

    length                       数组中元素的数量



字符串属性

     length                      字符串中多少个字符



元素的通用属性

     el.textContent                         // 元素的内容

  el.innerHTML                        // 元素的通用属性，获取 HTML 内容或添加 HTML 内容到页面上。

     el.className                           // 元素的通用属性，设置或返回元素的 class 属性的值

     el.innerText                           // 仅获取设置文本

     el.id                                  // id属性

     el.value                               // 文本节点的值

     el.disabled                            // 为true，使控件失效
