遍历目录

    $dir = opendir(...);               // 打开所浏览的目录

    $file = readir($dir);              // 从目录中读取文件

    closedir($dir);                    // 关闭该目录

    rewinddir($dir);                   // 将所读的文件恢复到开始的目录

    $dir = dir("/uploads/");           // dir类， Return an instance of the Directory class

    $dir->read();

    $dir->close();

    $dir->rewind();

    scandir(...);                      // 以字母表升序或降序将文件名保存在数组中

    glob(...);                         // 找到匹配模式的所有文件，可以用 glob(path/*.txt) 来找 path 下的所有.txt结尾的文件
