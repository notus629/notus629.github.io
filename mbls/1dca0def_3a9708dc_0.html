简单步骤

    // 方式一，使用 DSN 建立一个数据库连接（DSN：数据源名称）
    $dsn = "mysql:dbname=webuser;host=localhost;charset=utf8";
    $db = new PDO ($dsn, $username, $password);                                                                   
    
    // 方式二， PDO 建立一个MySQL连接 
    $db = new PDO("mysql: host=localhost; dbname=library", "pertermac", "abc123");                 
    
    // 向服务器发送 SQL 命令
    $db->query("UPDATE books SET authorid=4  WHERE pub_year=1982");
