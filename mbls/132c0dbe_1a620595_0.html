Dom 函数

    调用方式：document.function(xxx);


    getElementsByTagName()        获取指定标签名的对象集合

    getElementsByName()           获取带有指定名称(name属性)的对象集合

    querySelector()               获取指定 CSS 第一个元素

    querySelectorAll()

    getElementById()              获取指定 id 的第一个对象元素


    hasAttribute()                判断标签是否含有指定属性

    setAttribute()                添加指定的属性，并为其赋值

    removeAttribute()             删除标签指定的属性

    createAttribute()             创建一个指定名称的属性

    getAttribute()                获取指定属性名的属性值

    getAttributeNode()            获取属性节点


    removeChild()                 删除指定元素的某个指定的子节点

    appendChild()                 向节点末添加一个子节点

    insertBefore()                在指定的子节点之前插入新的子节点


    createElement()               创建元素节点

    createTextNode()              创建新文本节点



属性

    innerHTML                      设置或返回标签之间的 HTML 内容

    childNodes                     获得指定元素的子节点集合

    className                      设置或获取元素的 class 属性

    innerText                      设置或获取节点的文本

    lastChild                      获取最后一个子节点

    firstChild                     获取第一个子节点

    nextSibling                    返回同级节点相邻的下一个节点

    previousSibling                返回同级节点相邻的上一个节点

    parentNode                     返回元素的父节点

    nodeName                       获取标签节点的名称
