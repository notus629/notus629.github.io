<?php
/**
 * 实例描述：从 FTP 服务器下载一个文件到本地
 */
 
$curlobj = curl_init();
curl_setopt($curlobj, CURLOPT_URL, "ftp://localhost/fordownload.txt");
curl_setopt($curlobj, CURLOPT_HEADER, 0);
curl_setopt($curlobj, CURLOPT_RETURNTRANSFER, TRUE);
// 若下载超过 300 秒，则超时失败
curl_setopt($curlobj, CURLOPT_TIMEOUT, 300);
// 设置 FTP 用户名密码
curl_setopt($curlobj, CURLOPT_USERPWD, 'leo:567890');
 
// 下载保存的本地文件，先以写的方式打开
$outfile = fopen('ftpdest.txt', 'wb');
// 设置下载保存的文件
curl_setopt($curlobj, CURLOPT_FILE, $outfile);
 
$rtn = curl_exec($curlobj);
fclose($outfile);
if(!curl_errno($curlobj)){
    // $info = curl_getinfo($curlobj);
    // print_r($info);
    echo "Download successful!, RETURN code: ". $rtn;
} else {
    echo "curl error: ". curl_error($curlobj);
}
