<?php
 
// 静态缓存的生成、获取、删除，包含失效时间
// file1.php
class File {
 
    private $_dir;
 
    const EXT = '.txt';
 
    public function __construct(){
        $this->_dir = dirname(__FILE__).'/cache/';
    }
 
    // 缓存失效时间 $cacheTime 和 $data 一并存入文件
    public function staticCache($key, $data='', $cacheTime=0){
 
        $dir = $this->_dir;
 
        $filename = $dir.$key.self::EXT;
 
        // $data 有值则为设置
        if( $data !== ''){
 
            // 为 null 则删除缓存
            if ( is_null($data) ){
                return @unlink($filename);
            }
 
            if ( !is_dir( $dir ) ){
                mkdir($dir, 0777);
            }
 
            $cacheTime = sprintf("%011d", $cacheTime);
 
            return file_put_contents( $filename, $cacheTime.json_encode($data) );
        }
 
        // $data 无值，则为获取数据
        if(!is_file($filename)){
            return False;
        }
        $contents = file_get_contents($filename);
        $cacheTime = (int)substr($contents, 0, 11);
        $value = substr($contents, 11);
        if( $cacheTime != 0 && $cacheTime + filemtime($filename) < time() ){
            unlink($filename);
            return False;
        }
        return json_decode( $value, TRUE );
    }
}
