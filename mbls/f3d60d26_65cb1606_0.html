<head>

    <title>完成列表综合示例</title>
    <script type="text/javascript" src="js/jquery.js"></script>
    <script type="text/javascript">
        $(function(){
            //SETUP
            var $list, $newItemForm, $newItemButton;
            var item = '';
            $list = $('ul');
            $newItemForm = $('#newItemForm');
            $newItemButton = $('#newItemButton');
 
            $('li').hide().each(function(index){
                $(this).delay(450 * index).fadeIn(1600);
            });
 
            function updateCount(){
                var items = $('li[class!=complete]').length;
                $('#counter').text(items);
            }
 
            updateCount();
 
            // SETUP　FROM FOR NEW ITEMS
            $newItemButton.show();
            $newItemForm.hide();
 
            $newItemButton.on('click', function(){
                $newItemButton.hide();
                $newItemForm.show();               
            });
 
            // ADDING A NEW LIST ITEM
            $newItemForm.on('click', function(e){
                item = $('input:text').val();
                if(item !== ''){
                    $list.append('<li>' + item + '</li>');
                    $newItemButton.show();
                    $newItemForm.hide();   
                    $('input:text').val('');
                    updateCount();
                }
                e.preventDefault();
            });
 
            // CLICK HANDING -USES DELEGATION ON ul ELEMENT
            $list.on('click', 'li', function(){
                var $this = $(this);
                var complete = $this.hasClass('complete');
                if(complete === true){
                    $this.animate({
                        opacity: 0.0,
                        paddingLeft: '+180'
                    }, 500, 'swing', function(){
                        $this.remove();
                    });
                } else {
                    item = $this.text();
                    $this.remove();
                    $list.append('<li class="complete">' + item + '</li>').hide().fadeIn(3000);
                    updateCount();
                }
            });    
        });
    </script>
    <style type="text/css">
        h3 {
            text-align: center;
        }
        .complete {
            background-color: #ccc;
        }
        #counter {
            color: red;
            border: 1px dotted green;
            font-size: small;
        }
    </style>
