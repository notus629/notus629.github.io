创建表

    mysql > CREATE TABLE tablename ( column_name_1  column_type_1  constraints,
                                     column_name_2  column_type_2  constraints,
                                     ...
                                     column_name_n  column_type_n  constraints);



Constraints

    NOT NULL

        必须有一个值，若没有指定，可为空（NULL）


    AUTO_INCREMENT

        在整数字段中使用，若为空，则自动产生一个唯一的标识符

        比本列中现在的最大值更大

        此列必须是索引列


    PRIMARY KEY

        该列是表的主键，输入必须唯一

        MySQL 将自动索引该列

        PRIMARY KEY(orderid, isbn)                 //多列主键
