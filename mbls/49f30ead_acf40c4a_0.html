示例 - Ajax 加载 HTML



ajax.html             
    <!DOCTYPE html>
    <html>
    <head>
        <meta charset="utf-8">
        <title>使用 Ajax 加载 html</title>
        <script>
            var xhr = new XMLHttpRequest();
            xhr.onload = function(){
                if(xhr.status === 200){
                    document.getElementById('box').innerHTML = xhr.responseText;
                }
            }
            xhr.open('GET', 'data/data.html', true);
            xhr.send();
        </script>
        <style>
            #box{
                border: 2px dotted fuchsia;
                width: 50%;
                height: 300px;
            }
        </style>
    </head>
    <body>
    <h3>Notus</h3>
    <div id="box"></div>
    <h3>Notus</h3>
    </body>
    </html>


data/data.html           
    <!DOCTYPE html>
    <html lang="en">
    <head>
        <meta charset="UTF-8">
        <title>data</title>
    </head>
    <body>
    <h3>Hi Notus!</h3>
    </body>
    </html>
