禁用 Sublime 自动更新提示

    Preferences->Settings, 大括号中加入
        "ignored_packages":[],
        "update_check":false



打开自动完成与自动提示

    Preferences->Settings, 大括号中加入
        "auto_complete": true,
        "auto_match_enabled": true



自动换行

    Preferences->Settings, 大括号中加入
        "word_wrap": true
