str 系列

    // 返回字符串的长度
    strlen($str);

    // 返回子串第一次出现的位置（从 0 开始），否则 false
    strpos($_POST['email'], '@');      


    // 从某个位置开始抽取字符串的一部分 ($start, $length 为负则从末尾-1开始索引，不取末尾)
    substr($string, $start, $length); 

    // 默认返回出现 $needle (包含本身）及其后面的字符串，否则 false
    // 注意：$needle 不一定为单个字符，在 $haystack 中查找 $needle 中的每一个字符
    strstr($haystack, $needle, $isbeforeneedle=false);
    stristr(...) // case insensitive version


    // $new_str 替换 $old_str 中 $start 开始的 $len 长度，均为0则插入
    substr_replace($old_str, $new_str, $start, $len);

    // Replace all occurrences of the search string with the replacement string
    mixed str_replace ( mixed $search , mixed $replace , mixed $subject [, int &$count ] )


    // Split a string by delimiter
    arr explode($delimiter, $string [, $limit])          

    // 将数组以空格分隔组成字符串
    arr implode(' ', $arr);           

    // 将字符串重复 n 次后返回
    str_repeat('hi', 100);

    // 按字节(符)反转
    strrev($str)                                         

    // 去除 $date 前后的 " 符号
    $date = trim($date, '"');                

    // Converts a string to an array.
    array str_split ( string $string [, int $split_length = 1 ] );            



示例

    ## 输出：  olleh
    print strrev('hello');
