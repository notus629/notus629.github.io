示例 - wampserver3.0.6 环境

    在 httpd.conf 配置文件中引入了 virtual hosts 配置文件（httpd-vhosts.conf）

        # Virtual hosts
        Include conf/extra/httpd-vhosts.conf


    在 httpd-vhosts.conf 文件中，定义如下

        # Virtual Hosts
        #
        ## 虚拟主机 1 配置
        <VirtualHost *:80>
            ServerName localhost
            DocumentRoot c:/wamp64/www

            ## 对虚拟主机目录的配置
           <Directory  "c:/wamp64/www/">
                Options +Indexes +Includes +FollowSymLinks +MultiViews
               AllowOverride All
               Require local
           </Directory>
        </VirtualHost>

        ## 虚拟主机 2 配置
        <VirtualHost *:80>
            ServerName test.open.bugber.com
            DocumentRoot d:/www

            <Directory  "d:/www/">
               Options +Indexes +Includes +FollowSymLinks +MultiViews
               AllowOverride All
               Require local
           </Directory>
        </VirtualHost>
