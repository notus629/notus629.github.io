连接

        var conn = @mysql_connect(host, name, pwd);

        @mysql_select_db(db_leo, conn);                                    //选择数据库

        mysql_query("set name gb2312");                                    //设置编码

        mysql_query($sql, conn);                                                  //查询

        mysql_errno();                                                                  // Returns the numerical value of the error message from previous MySQL operation
        mysql_error();                                                                   // Returns the text of the error message from previous MySQL operation

        @mysql_num_rows(result);                                               //返回结果result 的行数

        mysql_fetch_array(result, MYSQL_ASSOC);                      //取单条记录

        @mysql_query($sql);
        @mysql_affected_rows();

        mysql_free_result(result)

        mysql_close(conn);
