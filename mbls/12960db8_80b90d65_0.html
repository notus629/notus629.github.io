// 得到脚本或图片等文件的绝对路径
function getfullpath($var = '') {
    $httppath = "http://".$_SERVER['SERVER_NAME']."/phppractice/projects/shoppingcart";
    $ospath = $_SERVER['DOCUMENT_ROOT']."/phppractice/projects/shoppingcart";
//  $projpath= $_SERVER['DOCUMENT_ROOT']."/phppractice/projects/shoppingcart";
    switch ($var) {
        case "admin.php":
        case "insert_book.php":
        case "login.php":
        case "logout.php":
            $path = $httppath."/admin/".$var;
            break;
        case "admin_fns.php":
        case "book_fns.php":
        case "book_sc_fns.php":
        case "data_valid_fns.php":
        case "db_fns.php":
        case "order_fns.php":
        case "output_fns.php":
        case "user_auth_fns.php":
            $path = $httppath."/functions/".$var;
            break;
        case "imagespath":
            $path = $httppath."/images";
            break;
        case "checkout.php":
        case "process.php":
        case "purchase.php":
        case "show_cart.php":
            $path = $httppath."/shopcart/".$var;
            break;
        case "index.php":
        case "show_book.php":
        case "show_cat.php":
            $path = $httppath."/".$var;
            break;
        default:
            $path = "";
            break;
    }
    return $path;
}
