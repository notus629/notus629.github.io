wampserver3.0.6 示例

    在 httpd.conf, 中包含了定义 virtual hosts 的文件，配置如下

        # Virtual hosts
        Include conf/extra/httpd-vhosts.conf


    在 httpd-vhosts.conf 文件中，定义如下

        # Virtual Hosts
        #

        <VirtualHost *:80>
         ServerName localhost
         DocumentRoot c:/wamp64/www
        <Directory  "c:/wamp64/www/">
         Options +Indexes +Includes +FollowSymLinks +MultiViews
         AllowOverride All
         Require local
        </Directory>
        </VirtualHost>

        <VirtualHost *:80>
         ServerName test.open.bugber.com
         DocumentRoot d:/www
         <Directory  "d:/www/">
         Options +Indexes +Includes +FollowSymLinks +MultiViews
         AllowOverride All
         Require local
        </Directory>
        </VirtualHost>
        #
