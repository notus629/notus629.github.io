    curl_setopt($curlobj, CURLOPT_TIMEOUT, 300);

    $outfile = fopen('ftpdest.txt', 'wb');

    curl_setopt($curlobj, CURLOPT_FILE, $outfile);
 
    $rtn = curl_exec($curlobj);
    fclose($outfile);
