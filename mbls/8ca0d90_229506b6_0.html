改变数据库中表的结构

    ALTER  TABLE  [IGNORE]  table_name  alteration  [, alteration ... ]

        允许每个语句实现多次修改


ALTER TABLE 语句可能完成的修改

        ADD  [ COLUMN ]  column_description  [ FIRST | AFTER  column ]

        ADD  [ COLUMN ] (column_description, column_description, ... )

        ADD  INDEX [ index ] (column, ... )                                              ;为指定列添加索引

        ADD  [ CONSTRAINT  [ symbol ] ]  PRIMARY  KEY (column, ... )

        ADD  UNIQUE  [ CONSTRAINT CONSTRAINT [ symbol ] ] [index] (column, ...)

        ADD  [ CONSTRAINT  [ symbol ] ]  FOREIGN  KEY  [ index ] ( index_col, ...) [ reference_definition ]

        ALTER [ COLUMN ] column { SET DEFAULT  value | DROP DEFAULT}                      ;添加或删除特定的默认值

        CHANGE [ COLUMN ] column  new_column_description 

        MODIFY [ COLUMN ] column_description

        DROP [ COLUMN ] column

        DROP  PRIMARY  KEY

        DROP  INDEX  index 

        DROP  FOREIGN KEY  key

        DISABLE  KEYS

        RENAME [ AS ] new_table_name

        ORDER  BY  col_name

        CONVERT  TO  CHARACTER  SET  cs COLLATE  c

        [ DEFAULT ] CHARACTER  SET  cs  COLLATE  c

        DISCARD  TABLESPACE

        IMPORT  TABLESPACE
