是否为空
        
    // 变量不存在, 变量声明了但未赋值，设置为 NULL 或 unset 后，
    // - 返回 TRUE。其余返回 FALSE
    is_null();                         

    // 值是等于 FALSE 值或不存在, 返回 TRUE    
    // 等价于 !isset($var) || $var == false
    // empty 值： "", 0, 0.0, "0", NULL, FALSE, array(), 
    // - $var(declared, but without a value)
    // 注：对于空格字符串： "      "，返回 FALSE, 即非空
    empty();                           
        
    // Determine if a variable is set and is not NULL
    isset();
