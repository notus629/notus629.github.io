typeof 

    返回其类型，对于基本类型和函数类型方便，其它不方便

    typeof NaN     // "number"

    typeof null    // "object", 历史原因，兼容问题

    typeof 100     // "number"



instanceof

    判断对象的类型

    obj instanceof object        // 期待左操作数为对象，期望右操作数必须是一个函数对象或函数构造器

    不同的window或iframe间的对象类型检测不能使用instanceof



Object.prototype.toStirng

    Object.prototype.toString.apply([]);  === "[object Array]";

    Object.prototype.toString.apply(function(){});  === "[object Function]";
