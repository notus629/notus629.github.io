Outline:

        构造函数: __construct();                               // 可带参数，
        析构函数: __destruct();                                // 析构函数，不能有任何参数
        __get($name){return $this->$name;}                                        // 针对私有属性，访问函数
        __set($name, $value){ $this->$name = $value;}

        extends                                                           // 实现继承
        重载
        paraent                                                           // 调用父类的最初版本
        final 关键字                                                   // 函数不能被重载，在类前面则类不能被继承

        interface Displayable {                                    // 接口
                function display();
        }
        
        implements                                                      // 实现接口， 一个类可以实现一个或多个接口

        per-Class 常量：const定义
        ::操作符来调用

        静态方法：static function func(..){...}
        :: 调用

        instanceof                                                          // 检查对象的类型，是否是特定的类，是否从某个类继承过来，或是否实现了特定的接口
        延迟静态绑定
        clone 复制已有对象
        abstract 抽象类、抽象方法

        __call()方法                                                      // 来实现方法的重载
        __autoload()函数                                              //在类声明之外声明，在实例化一个还没有被声明的类时自动调用

        iterator迭代器
                要迭代的类实现 IteratorAggregate接口
                并且定义一个能返回该迭代类实例的getIterator方法
                该类必须实现 Iterator 接口

        在类中实现 __toString()函数                                        // 尝试打印该类时，会调用此函数
        var_export()函数                                                            // 打印出类中的所有属性值

        反射 Reflection API                                                         // Reflection类，通过访问已有类和对象来找到类和对象的结构和内容的能力
