表单事件

    submit        表单被提交时，会触发 form 上的 submit 事件

    change        某些表单元素的状态发生变化时触发：下拉列表中选择一个选项，选中一个单选按钮，选中或取消一个复选框

    input         用于 input 和 texterea 元素

    focus/blur    也用于其它元素

<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>使用表单事件</title>
</head>
<body>
    <form action="a.php" method="post">
        <p>
        <label> 选择一门编程语言：<span id="hint"></span><br/>
            <select>
                <option>Java</option>
                <option>PHP</option>
            </select>
        </label>
        </p>
        <p><input type="checkbox" id="terms">我同意此协议</p>
        <p><span id="hint2"></span></p>
        <input type="submit" value="提交">
    </form>
    <script type="text/javascript">
        function submitCheck(e){
            var term = document.getElementById('terms');
            var hint2 = document.getElementById('hint2');
            if( !term.checked){
                hint2.textContent = "必须同意协议!";
                e.preventDefault();
            }
        }
 
        function hintMsg(){
            var hint = document.getElementById('hint');
            hint.textContent = this.options[this.selectedIndex].value;
        }
 
        var form = document.getElementsByTagName('form')[0];
        var select = document.getElementsByTagName('select')[0];
 
        form.addEventListener('submit', submitCheck);
        select.addEventListener('change', hintMsg);
    </script>
 
</body>
