<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>录音界面</title>
</head>
<body>
    <p>名称：<span id="notetitle">note</span></p>
    <p><input type="text" id="notename"></p>
    <button data-state="record">录音</button>
</body>
<script type="text/javascript">
    var notetitle = document.getElementById("notetitle");
    var notename = document.getElementById("notename");
    var btn = document.querySelector('button');
 
    function writeLabel(e){
        if(!e){
            e = window.event;
        }
        var target = e.target || e.srcElement;
        notetitle.textContent = target.value;
    }
 
    function recorderControls(e){
        if(!e){
            e = window.event;
        }
        var target = e.target || e.srcElement;
 
        if(e.PreventDefault){
            e.PreventDefault();
        } else {
            e.returnValue = false;
        }
 
        switch(target.getAttribute('data-state')){
            case 'record':
                record(target);
                break;
            case 'stop':
                stop(target);
                break;
            default:
                break;
        }
    }
 
    function record(target){
        target.textContent = '停止';
        target.setAttribute('data-state', 'stop');
    }
 
    function stop(target){
        target.textContent = '录音';
        target.setAttribute('data-state', 'record');
    }
 
    if(addEventListener){
        notename.addEventListener('keyup', writeLabel);
        btn.addEventListener('click', recorderControls);
    } else {
        notename.attachEvent('onkeyup', writeLabel);
        btn.attachEvent('onclick', recorderControls);
    }
 
</script>
</html>
