<?php
    header("location: index.php");          // 跳转到index.php, header 函数前不能有输出    
 
    echo"<script>window.history.go(-1);</script>";   // 利用JavaScipt, 跳转到浏览历史的上一页
  
   // 按钮单击退回 也可以用 back(-1)
    echo "<button onclick=\"return window.history.go(-1);\">返回</button>"; 
 
    // 利用 meta 标签进行延时跳转, 延时1秒, 只能在head元素中
    echo "<meta http-equiv=\"refresh\" content=\"2; url=\"http://bing.com/\" />";
?>
