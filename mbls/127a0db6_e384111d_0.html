修改默认字符集为 utf8 [1]

    定位到 my.ini 文件，添加以下内容

        [client]
        default_character_set=utf8

        [mysqld]
        ; default-character-set=utf8
        character-set-server=utf8
        collation-server = utf8_general_ci        ; 校验字符集为 utf8_general_ci

        [mysql]
        default_character_set=utf8
