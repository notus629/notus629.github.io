    file_exists(...);                  // 文件是否存在
    dirname(...);                      // 返回路径的目录部分
    basename(...);                     // 返回路径的文件名称部分

    rmdir(...);                        // 删除目录
    umask(...);                        // 设定umask，会返回旧的umask
    touch(...);                        // 创建一个文件，或修改文件上次被修改的时间
    unlink(...);                       // 删除一个文件
    copy(...);                         // 复制文件                     
    rename(...);                       // 重命名文件，无 move 函数，此函数来完成
