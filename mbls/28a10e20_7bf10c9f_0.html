超级全局变量

    $_COOKIE['mycookie']                // 获取/设置 cookie 中名称为 'mycookie' 的值, 

    $_SESSION[' ']                      // 会话全局变量

    $_ENV                               // 系统的环境变量数组

    $_FILES                             // 与文件上传相关的变量数组

    $_REQUEST                           // 所有用户输入的变量数组，包括 $_GET, $_POST, 



$_SERVER

    $_SERVER['SERVER_NAME']             // 服务器主机名（域名）

    $_SERVER['SCRIPT_NAME']             // 即脚本的文件名

    // The filename of the currently executing script， 不包括域名（主机名）
    $_SERVER['PHP_SELF']                

    $_SERVER['HTTP_REFERER']            // 引导用户到当前页的前一页的地址
