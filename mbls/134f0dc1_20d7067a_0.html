随机数

    // 返回 min 到 max 之间的随机整数, 包括 max
    // php7.1.0 后为 mt_rand 别名
    // 默认 0 到 getrandmax();
    int rand( [int min, int max] );    

    // Generate a better random value
    int mt_rand ($min, $max)           

    // 返回最大可能的随机数值
    int getrandmax(void)              


    // seed the random number generator
    // or with a random value if no $seed is given
    // php7.1.0 后, 同 mt_srand
    void srand ([ int $seed ] );
    void mt_srand([ int $seed ]);


    // php7
    // Generates cryptographically secure pseudo-random bytes
    string random_bytes ( int $length );

    // php7
    // Generates cryptographically secure pseudo-random integers
    // $min 须 >= PHP_INT_MIN, $max 须 <= PHP_INT_MAX
    int random_int ( int $min , int $max )

    // PHP 5 >= 5.3.0, PHP 7
    // $crypto_strong, 为一个引用变量，可以判断是否是密码学安全的    
    string openssl_random_pseudo_bytes ( int $length [, bool &$crypto_strong ] );


    // Gets a prefixed unique identifier based on the current time in microseconds.
    // prefix 为空则返回 13 字符，more_entropy 为 TRUE，则返回 23 字符
    string uniqid ([ string $prefix = "" [, bool $more_entropy = false ]] )
