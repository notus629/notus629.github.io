支持旧版本IE

    在 IE5~IE8 中，不支持 addEventListener, 但有类似的函数: attachEvent

        // 注意：此处事件名前要加 "on" 
        el.attachEvent("onblur", function(){
            checkUsername(5);
        }, false);



同时支持各种浏览器

    // 浏览器支持 addEventListener, 则返回 true
    if(el.addEventListener){ 
        el.addEventListener('blur', function(){
            checkUsername(5);
        }, false);
    } else {
        el.attachEvent('onblur', function(){
            checkUsername(5);
        }, false);
    }
