七、了解数据库的SQL语句操作
         SQL (Structured Query Language)
       DDL
               create database 库名        //创建库
               create table [库名.]表名          //创建表 使用了use 库名，则可省略表名
               drop database 库名        //删除库
               drop table 表名        //删除表
         DML
                 insert into users(id, name) values('1', 'zhangsan');                //插入
                 update users set name='aa', age='10' where id='1';        //更新
                 delete from 表名 where id='2';                //删除
         DQL        数据查询语言
                 select * from 表名;        //查询
         DCL        数据控制语句，管理数据库的语言，包含管理权限及数据的更改，用工具去执行
                \s看状态
               show databases 看所有库
               show tables 看所有表
               desc 看表结构
               show variables 看配置文件中的变量和值
使用数据库步骤
       1. 执行SQL语句，连接到数据库服务器
       2. 创建数据库xsphpdb
               create database xsphpdb;
               删除:drop database xsphpdb;
               在data\下建一个目录也就是建一个库
               如果不存在创建：create database if not exists xsphpdb;
               如果存在才删除：drop database xsphpdb if exists xsphpdb;
               库中创建表：
                       create table xsphpdb.users(id int, name char(30), age int, sex char(3));
       3. 选择一个库做为默认的数据库
               使用其中的表就不用再加库.表名，直接用表名即可
               use xsphpdb;
               查看users表结构：desc users;
               删除users表：drop table if exists users;
               创建表：create table users(id int, name char(30), age int, sex char(3)) if not exists users;
       4. 增删改查
               insert into users values(1, 'zhangsan', 10, 'man'); //值按顺序插入
               因为是弱类型，若全部用单引号，会自动去转换成相应的类型
               若只插入部分字段，出错
               在全插入和只插入部份，可以先给出字段：可以插入部分，且不用考虑顺序。
                       insert into users(id, name, age) values('10', 'wangwu', '33');        //正确
               查看数据
                       select * from users;
               更新语句
                       update users set name='lili', age='21' where id='2';
               删除
                       delete from users where id=2;
       帮助的使用
               使用mysql安装后自带的帮助文档

               ? contents        //看帮助的内容
