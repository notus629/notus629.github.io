<?php
 
// 单例模式，只能拥有一个实例
 
class Db
{
    private static $_instance;
 
    private $_dbConfig = array(
            'host' => '127.0.0.1',
            'user' => 'root',
            'password' => '',
            'database' => 'video',
        );
 
    private static $_connectSource;
 
    private function __construct(){
 
    }
 
    public static function getInstance(){
        if(!self::$_instance instanceof self){
            self::$_instance = new self();
        }
        return self::$_instance;
    }
 
    public function connect(){
        if(!self::$_connectSource){
            self::$_connectSource = mysqli_connect($this->_dbConfig['host'],
                $this->_dbConfig['user'],
                $this->_dbConfig['password'] );
 
            if(!self::$_connectSource){
                throw new Exception('mysql connected error'.mysqli_error(self::$_connectSource));
                //die('mysql connected error'.mysqli_error(self::$_connectSource));
            }
 
            mysqli_select_db(self::$_connectSource, $this->_dbConfig['database']);
        // 设置字符编码
            mysqli_query(self::$_connectSource, 'set names UTF8');          
        }
        return self::$_connectSource;
    }
}
