创建多个对象：构造函数语法

    先建一个模板，即用来创建对象的函数

        function Hotel(name, rooms, booked){
            this.name = name;
            this.rooms = rooms;
            this.booked = booked;
            this.checkAvailability = function(){
                return this.rooms - this.booked;
            };
        }



使用构造函数，创建对象的实例
               
    var quayHotel = new Hotel('Quay', 40, 20);
