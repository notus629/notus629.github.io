<?php
 
function run_time(){
    list($msec, $sec) = explode(' ', microtime());
    return (float)$msec + (float)$sec;
}
 
// 开始时间
$start_time = run_time();
 
/* 运行 PHP 代码段 */
$time1 = strtotime(date('Y-m-d'));
$time2 = strtotime(date('2020-12-31'));
$sub = ($time2 - $time1) / (60 * 60 * 24);
echo "距离2020年12月31日还有<font color=red>$sub</font>天!!!";   
/* *************** */
 
// 结束时间
$end_time = run_time();
$duration = $end_time - $start_time;
?>
<br />
倒计时程序的运行时间为<font color="blue"><?php echo $duration; ?></font>秒
