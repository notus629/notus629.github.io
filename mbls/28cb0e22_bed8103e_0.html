PHP正则表达式函数

    preg_match();                       // Perform a regular expression match （once)

    preg_match_all(...);                // Perform a global regular expression match

    // Searches $subject for matches to $pattern and replaces them with $replacement.
    // $pattern can be either a string or an array with strings.
    // If both pattern and replacement parameters are arrays, each pattern will be replaced by the replacement counterpart.
    // $subject may be an array.
    // $limit, The maximum possible replacements 
    // &$count, the number of replacements done. 
    mixed preg_replace ( mixed $pattern , mixed $replacement , mixed $subject [, int $limit = -1 [, int &$count ]] );

    
    // $callback will be called and passed an array of matched elements in $subject
    // The callback should return the replacement string.
    // callback signature: string handler ( array $matches )
    mixed preg_replace_callback ( mixed $pattern , callable $callback , mixed $subject [, int $limit = -1 [, int &$count ]] );


    // Split string by a regular expression
    array preg_split ( string $pattern , string $subject [, int $limit = -1 [, int $flags = 0 ]] )
    
    // Split string into array by regular expression
    // DEPRECATED in PHP 5.3.0, and REMOVED in PHP 7.0.0.
    array split ( string $pattern , string $string [, int $limit = -1 ] )


    ereg(...);                          // 正则表达式匹配，5.3.0起不赞成，7.0版本移除             
    ereg_replace(...);                  // 正则表达式替换，5.3.0里不赞成，7.0版本废除 

    eregi(...);                         // 不区分大小写版，5.3.0里不赞成，7.0版本废除 

    eregi_replace(...);                 // 不区分大小写版，5.3.0里不赞成，7.0版本废除
