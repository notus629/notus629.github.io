        字符串连接和分割
                explode(separator, input, [, limit]);
                implode($sep, $str), join()       // 与explode相反效果
                strtok($input, $separator);               
                // $separator 也是字符串，每次从sep中取一个做为分隔符。得到第一个令牌：用两个参数。
                // 后面每次取令牌序列只用一个参数，即$sep. 重置指针用两个参数再次调用。
                substr($str, $start [, $len]);    // 取从$start开始的 $len个字符，忽略$len,取整个， 负数表示从末尾开始

        字符串匹配和替换子串
                strstr($haystack, $needle)/strchr()       //返回关键字及其后的字符串
                stristr()                                 // 不分大小写
                strrchr()                                 // 返回关键字及其最后一次出现到字符串末尾的字符串
                strpos(haystack, needle, [offset])        //  返回的数代表被搜索字符串第一次出现目标子串的位置（0开始）                 strrpos(); 
                str_replace(needle, new_needle, haystack [, count])     //替换，并返回结果，count为替换次数
                substr_replace(str, replacement, start, len);           // 在给定位置中查找和替换字符串中特定的子字符串
                
        正则表达式的介绍
                POSIX 和 Perl 风格（PCRE）
                
                字符集和类
                        .                    // 换行符 \n 外的任意一个字符
                        [ ]                  // 匹配方括号中的字符集合中的一个字符，[a-zA-Z]：范围， 或[aeiou]: 列表
                        [^a-z]               // 不属于某个集
                        [[:alnum:]] 等       // POSIX风格的正则表达式的字符类

                重复
                        *                    // 表示这个模式可以被重复0次或更多次
                        +                    // 表示这个模式可以被重复1次或更多次
                        ？                   // 子模式0次或1次
                
                子表达式
                        (very)*              // very被重复0次或多次
                        (very ){1, 3}        // 计数

                 ^bob                 // 定位到开始
                 com$                 // 定位至末尾
                 com | edu | cn       // 分支

                正则表达式查找子串

                        // 查找正则表达式pattern的字符串，用replacement来替换
                        eregi_replace();
                        array split(pattern, search, max);       // 按正则表达式模式分割，返回数组，指定最多匹配的个数
