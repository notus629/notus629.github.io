dpkg                                      

    -i, --install                       ## 安装，会删除旧版本
    --force                             ## 强制安装，忽略一切依赖和兼容问题

    $ dpkg -l | grep openssh            ## 查看是否已安装 openssh

    -S, --search                        ## 查看已安装的软件向系统中复制了哪些文件
    $ dpkg -S openssh                   ## 查看哪些文件是 openssh 软件带来的

    -r, --remove                        ## 卸载软件包
