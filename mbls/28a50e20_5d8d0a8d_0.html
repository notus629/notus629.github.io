设置 root 密码

    用 root 登录进入 phpMyAdmin->权限->修改密码

    在 phpmyadmin\ 下找到 config.inc.php 文件（xampp1.7.3, windows），定位到相应位置，修改如下

    /* Authentication type and info */
    $cfg['Servers'][$i]['auth_type']            = 'http';
    $cfg['Servers'][$i]['user']                 = 'root';
    $cfg['Servers'][$i]['password']             = '567890';
    $cfg['Servers'][$i]['AllowNoPassword']      = false;
