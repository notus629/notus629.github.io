<?php
/**
 * 实例描述：下载网络上的一个 HTTPS 的资源
 */
 
$curlobj = curl_init();
// 设置访问（要下载的）网页的 URL
curl_setopt($curlobj, CURLOPT_URL, "https://ajax.aspnetcdn.com/ajax/jquery.validate/1.12.0/jquery.validate.js");
curl_setopt($curlobj, CURLOPT_RETURNTRANSFER, true);
 
// 设置 HTTPS 支持
 
// HTTPS 访问有时效性，为了做认证的时效性比较，需要设置
date_default_timezone_set('PRC'); 
// 终止从服务端进行验证，
// HTTPS 是用来保证服务器就是我们想要访问的服务器
// 不关心可以跳过此步骤，这样就可以下载到 HTTPS上的资源
curl_setopt($curlobj, CURLOPT_SSL_VERIFYPEER, 0);
 
$output = curl_exec($curlobj);
curl_close($curlobj);
echo $output;
