<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>遍历DOM-示例</title>
    <script type="text/javascript" src='js/jquery-3.1.1.min.js'></script>
    <script type="text/javascript">
        $(function(){
            var $h2 = $('h2');
            $('ul').hide();
            $h2.append('<a href="#">show</a>');
            // 点击 h2 元素的任意点，触发匿名函数
            $h2.on('click', function(){
                // 获得 h2 元素的下一个兄弟节点，即 ul 节点
                $h2.next()
                // 淡出
                .fadeIn(500)
                // 选取 ul 的子节点的中所有class属性包含 hot 的那些节点
                .children('.hot')
                // 给选中的节点增加一个 class 名称 complete
                .addClass('complete');
            // 将 h2 中的链接淡出
            $h2.find('a').fadeOut();
            });
        });
    </script>
    <style type="text/css">
        .hot {
            color: red;
        }
        .complete {
            border: 2px solid blue;
        }
    </style>
</head>
<body>

<ul>
    <li class="hot">PHP</li>
    <li class="hot">JavaScript</li>
    <li>Python</li>

</ul>
