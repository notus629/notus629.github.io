<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>test07</title>
  
</head>
<body style="margin-top:100px;margin-left:100px;">
    <input type="text" id="username" name="" placeholder="请输入用户名">
    <div id="feedback" style="color:red;"></div>
    <script type="text/javascript">
        var username = document.getElementById("username");
 
        if(username.addEventListener){
            username.addEventListener('focusout', function(){
                checkUsername(10);
            });
        } else {
            username.attachEvent('onfocusout', function(){
                checkUsername(10);
            });
        }
  
        function checkUsername(lengthLimit){
            var feedback = document.getElementById("feedback");
            if(username.value.length < lengthLimit){
                feedback.innerHTML = "至少"+ lengthLimit +"字符！"
            } else {
                feedback.innerHTML = '';
            }
        }
    </script>
</body>
</html>
