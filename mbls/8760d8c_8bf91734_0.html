<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8" />
    <title>目录递归遍历</title>
</head>
<body>
    <h1>列出指定目录下的所有文件（包含各级子目录下的文件）</h1>
    <p>## 打印目录名，但不打印 . 和 .. 两个特殊文件</p>
    <?php
 
        getfiles('./');
 
        // 输入 $path 格式为：目录名最后带'/'符号
        function getfiles($path){
 
            $files = scandir($path);
            foreach ($files as $file) {
                if($file == '.' || $file == '..'){
                    continue;
                }
                 
                // 构造子文件全路径名称
                $fullfile = $path.$file;
                 
                // 判断子文件是否为目录，是就递归
                if(is_dir($fullfile)){
                    $fullfile .= '/';
                    getfiles($fullfile);
                } 
                echo $fullfile.'<br />';
            }
        }
    ?>
</body>
</html>

</head>
<body>
    <h1>列出指定目录下的所有文件（包含各级子目录下的文件）</h1>
    <p>## 不打印目录名以及 . 和 .. 两个特殊文件</p>
    <?php
 
        getfiles('./');
 
        // 输入 $path 目录格式为：目录名最后带'/'符号
        function getfiles($path){
 
            $files = scandir($path);
            foreach ($files as $file) {
                if($file == '.' || $file == '..'){
                    continue;
                }
                 
                // 构造子文件全路径名称
                $fullfile = $path.$file;
                 
                // 判断子文件是否为目录，是就递归，否则输出
                if(is_dir($fullfile)){
                    $fullfile .= '/';
                    getfiles($fullfile);
                } else{
                    echo $fullfile.'<br />';
                }
            }
        }
    ?>
</body>
</html>
