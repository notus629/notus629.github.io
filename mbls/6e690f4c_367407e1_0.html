Date/Time 对象
               
    创建 Date 对象，使用构造函数语法

        // 默认存储当前日期，存储其他日期则在参数中指定
        // JS 中日期以数字存储，即从1970年1月1日到现在为止的毫秒数
        var today = new Date();

     
    以下 3 种设置格式设置日期

        // YYYY, MM, DD, HH, MM, SS
        var dob = new Date(1996, 11, 26, 15, 45, 55);  

        // MMM DD, YYYY HH:MM:SS      
        var dob = new Date ('Dec 26, 1996 15:45:55');        
                               
        var dob = new Date(1996, 11, 26);
               

    创建日期后，以下方法来设置和返回它代表的时间和日期

        getDate()/setDate()                返回/设置月份中的日期（1-31）

        getDay()                           返回星期几(0-6)
                       
        getFullYear()/setFullYear()        返回/设置年份（4位数字）
                       
        getHours()/setHours()              返回/设置小时(0-23)
                       
        getMiliseconds()/setMiliseconds()  返回/设置毫秒数(0-999)
                       
        getMinutes()/getMinutes()          返回/设置分钟(0-59)
                       
        getMonth()/setMonth()              返回/设置月份(0-11)
                       
        getSeconds()/setSeconds()          返回/设置秒(0-59)
                       
        // 从1970年1月1日0点00:00:00 UTC（世界标准时间）
        // 开始计算的毫秒数。之前的时间返回负数
        getTime()/setTime()                
                       
        getTimezoneOffset()          按分钟为本地时间返回时区偏差值
                       
        toDateString()               返回humanreadable日期,如 Wed Apr 16 1975
                       
        toTimeString()               返回适合人类阅读的时间
                       
        toString()                   返回表示特定日期的字符串
