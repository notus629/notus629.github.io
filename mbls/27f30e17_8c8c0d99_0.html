        sort(...);
        $products = array('a', 'b', 'c',...);                   // 创建数组， array(...), 语言结构，非函数
        =                                                       // 数组复制
        range(1, 10, 2), range('a', 'z'), range(1, 10)          //  返回数组
        
        $products[0] =  'Fuses';                   // 改变数组元素的内容
        $products[3] =  '...';                     // 若索引3不存在，增加一个元素到数组末尾
        $products[0] =  'Fuses';                   // 直接在添加第一个元素的同时，创建一个数组


        foreach 访问数组
                foreach ( $products as $current){ echo $current; ...}

        关联数组
                $prices = array('Tires' => 100, 'Oil' => 10, ...);         // 创建
                $prices ['oil'] = '...';                                   // 若索引不存在，为数组增加元素
                $prices['Tires'] = '....';                                 // 在添加第一个元素的同时，可以创建数组

        在循环语句中使用关联数组
                foreach( $prices as $key => $value){...} 
                while ( $element = each($prices) ){$element['key']; $element['value']...; }    
                while ( list($product, $price) = each($prices) ){...};
                reset($prices);                  // 当前元素重新设置到数组开始处

        排序
                sort($products);
                SORT_NUMERIC, SORT_REGULAR, SORT_STRING             

                asort($prices);                                      // 关联数组，根据数组的每个元素值进行排序
                ksort($prices);                                      // 关联数组，根据关键字排序
                rsort(), arsort(), krsort();                         // 反向排序

                usort($products, 'compare');                         // 用户定义的排序
                function compare($x, $y);                            // 自定义比较函数

                uasort(...), uksort(...);
                shuffle(...);                                        // 将数组各元素进行随机排序
                array_reverse(...);                                  // 给出原来数组的反向排序

        array_push(...);                        // 新元素添加到数组末尾
        array_pop(...);                         // 删除并返回数组末尾的一个元素
        count(...);                             // 统计数组中元素的个数
        $arr = explode(seprator, str, limit);   // 将str分成小块，以separator为断点，返回数组，limit限制块的个数

        在数组中浏览
                current($arr);                               // 返回当前元素
                each($arr);                                  // 先往前移一个元素, 再返回当前元素
                next($arr);                                  // 先返回当前元素，再往前移一个元素
                reset($arr);                                 // 返回指向数组第一个元素的指针，移到开头
                end($arr);                                   // 移到末尾，返回最后一个元素
                prev($arr);                                  // 往回移一个位置，再返回新的当前元素

                array_walk($arr, func, [$userdata]);         // 对数组每一个元素，应用自定义函数
                your_function($value, [$key], [$userdata]);  // 用户定义函数, 索引数组$key的值可忽略

        统计数组元素个数
                count();
                sizeof();
                array_count_values($array);           //统计每个特定值在数组中出现过的次数，返回包含频率表的关联数组

        将数组转换成标量变量
                extract($array, [extract_type], [prefix]);       
                // 针对非索引数组，转换后 则系统中出现名字为$key1, $key2的变量值为其关键的value; 
                // type: 如何处理冲突， prefix: 加个前缀；
