            top:0px;
            left:0px;
            /*定义视图被置于X轴的何处*/
            transform-origin: left;
        }
        .page0 {
            background: url(./images/0.jpg) no-repeat;
        }
        .page1 {
            background: url(./images/1.jpg) no-repeat;
        }
        .page2 {
            background: url(./images/2.jpg) no-repeat;
        }
        .page3 {
            background: url(./images/3.jpg) no-repeat;
        }
        .page4 {
            background: url(./images/4.jpg) no-repeat;
        }
        .page5 {
            background: url(./images/5.jpg) no-repeat;
        }
        .page6 {
            background: url(./images/6.jpg) no-repeat;
        }
 
        /* 左翻页 */
        .flippage{
            transform: rotateY(180deg); /* 沿着 Y 轴的  3D旋转 */
            transition: all 1s;   /* 动画时长 */
        }
 
        /* 右翻页  */
        .flipback{
            transform: rotateY(180deg); /* 沿着 Y 轴的  3D旋转 */
            transition: all 1s;   /* 动画时长 */
        }
    </style>
 
    <script type="text/javascript">
 
        var page = []; //定义个空数组，用于储存
     
        function initPages(){
            var pages = document.getElementsByTagName("div");
 
            // 循环设置当前 div 的层数
            for(var i = 0; i < pages.length; i++){
                pages[i].style.top = "150px";
                pages[i].style.left = "600px";
 
                // 顺序设置当前的 div 层数，越大就越在上面
                pages[i].style.zIndex = 0-i;
            }  
        }
 
        function flip(){
 
            // 抓取现有的 css
            var classes = event.target.className.split(" ");
 
            // 回退触发的条件，总是找到最后一个 class, 判断当前的 class 是什么
            // 然后进行转换
            if(classes[classes.length - 1] == "flippage"){
                classes[classes.length - 1] = "flipback";
            } else if(classes[classes.length - 1] == "flipback"){
                classes[classes.length - 1] = "flippage";
            } else {
                // 返回第一次的动画效果
                classes.push("flippage");   // 直接插入
            }
 
            // zIndex 设置层级
            // event.target 返回触发了事件的 DOM　元素
            event.target.style.zIndex *= -1; 
 
            // 拼接当前的类名然后插入，然后就能执行 css 的动画了
            event.target.className = classes.join(" ");
        }
    </script>
